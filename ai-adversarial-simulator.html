Core Attack Algorithm (Python + PyTorch)

import torch
import torch.nn.functional as F

class AdversarialAttacker:
    def __init__(self, model, epsilon=0.1):
        self.model = model
        self.epsilon = epsilon
    
    def fgsm_attack(self, data, target):
        # Fast Gradient Sign Method
        data.requires_grad = True
        output = self.model(data)
        loss = F.cross_entropy(output, target)
        loss.backward()
        
        # Create adversarial perturbation
        perturbation = self.epsilon * data.grad.sign()
        adversarial_data = data + perturbation
        return torch.clamp(adversarial_data, 0, 1)
    
    def pgd_attack(self, data, target, alpha=0.01, iterations=10):
        # Projected Gradient Descent
        perturbation = torch.zeros_like(data)
        
        for i in range(iterations):
            perturbation.requires_grad = True
            output = self.model(data + perturbation)
            loss = F.cross_entropy(output, target)
            loss.backward()
            
            # Update perturbation
            perturbation = perturbation + alpha * perturbation.grad.sign()
            perturbation = torch.clamp(perturbation, -self.epsilon, self.epsilon)
            perturbation = torch.clamp(data + perturbation, 0, 1) - data
            
        return data + perturbation
    
    def evaluate_robustness(self, test_loader):
        # Test model robustness against attacks
        correct = 0
        total = 0
        
        for data, target in test_loader:
            adversarial_data = self.fgsm_attack(data, target)
            output = self.model(adversarial_data)
            pred = output.argmax(dim=1)
            correct += pred.eq(target).sum().item()
            total += target.size(0)
        
        return correct / total
